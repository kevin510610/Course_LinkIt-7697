#include "SevSeg.h"
SevSeg sevseg;

void setup() {
  byte numDigits = 4;
  byte digitPins[] = {2, 3, 4, 5};
  byte segmentPins[] = {8, 9, 10, 11, 12, 13,14,15};
  byte hardwareConfig = COMMON_CATHODE;
  sevseg.begin(hardwareConfig, numDigits, digitPins, segmentPins);
  sevseg.setBrightness(90);
}

void loop() {
  static unsigned long timer = millis();
  static int A=0,B=0;
  if (millis() - timer >= 300) {
    timer += 300;
    A++; 
    
    if (A == 60) { 
      A=0;
      B++;
    }
    if(B==24){
      B=0;
    }
    sevseg.setNumber(A+B*100);
  }
  sevseg.refreshDisplay();
}

#include "SevSeg.h"
SevSeg sevseg; 

void setup() {
  pinMode(6,INPUT);
  pinMode(16,INPUT);
  byte numDigits = 4;
  byte digitPins[] = {2, 3, 4, 5};
  byte segmentPins[] = {8, 9, 10, 11, 12, 13,14,15};
  byte hardwareConfig = COMMON_CATHODE; 
  sevseg.begin(hardwareConfig, numDigits, digitPins, segmentPins);
  sevseg.setBrightness(90);
}

void loop() {
  static unsigned long timer = millis();
  static int A=0,B=0;
    if (millis() - timer >= 300) {
        timer += 300;
        if(digitalRead(16)==1){
          A++;  
          if (A == 60) { 
            A=0;
            B++;
          }
          if(B==24){
            B=0;
          }
        }
        else{
          A--;
          if(A == -1){
            A = 59;
            if(B==-1){
              B =23;
            }
            else{
              B--;
            }
          }
        }
        if(digitalRead(6))
        {
          A=0;
          B=0;
          }
        sevseg.setNumber(A+B*100);
      }
  sevseg.refreshDisplay();
}
