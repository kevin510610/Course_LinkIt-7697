#include <Keypad.h>
#define NOTE_1  262
#define NOTE_2  294
#define NOTE_3  330
#define NOTE_4  349
#define NOTE_5  392
#define NOTE_6  440
#define NOTE_7  494
#define NOTE_11  523
#define NOTE_22  587
#define NOTE_33  659
#define NOTE_44  689
#define NOTE_55  784
#define NOTE_66  880
#define NOTE_77  988
#define NOTE_111  1047
#define NOTE_222  1109
const byte ROWS = 4;
const byte COLS = 4;
char keys[ROWS][COLS] = {
 {'1','2','3','4'},{'5','6','7','8'},
 {'9','A','B','C'},{'D','E','F','G'}
};
byte rowPins[ROWS] = {5,4,3,2};
byte colPins[COLS] = {11,10,9,8};
Keypad keypad = Keypad(makeKeymap(keys),rowPins,colPins,ROWS,COLS);

int tune[]={NOTE_1,NOTE_2,NOTE_3,NOTE_4,NOTE_5,NOTE_6,NOTE_7,NOTE_11,NOTE_22,NOTE_33,NOTE_44,NOTE_55,NOTE_66,NOTE_77,NOTE_111,NOTE_222};  
void setup() {
  pinMode(12, OUTPUT);
  Serial.begin(9600);
}

void loop() {
  int x = keypad.getKey();
  if(x != NO_KEY){
    if(x>='1' && x<='9'){
      x=x-'1';
    }
    else if(x>='A' && x<='G'){
      x=x-'A'+9;
    }
    tone(12,tune[x],300);
    }
}

char *morse[]={"01","1000","1010","100",
  "0","0010","110","0000",
  "00","00111","101","0100",
  "11","10","111","0110",
  "1101","010","000","1",
  "001","0001","11","1001",
  "1011","1100"};
void setup() {
  pinMode(12,OUTPUT);
  Serial.begin(9600);
}
char chr,index;
char *ptr;
void loop() {
  if(Serial.available()){
    chr=Serial.read();
    Serial.println(chr);
    if((chr-'A')>=0 && (chr-'Z')<=0){
      index=chr-'A';
      ptr=morse[index];
      while(*ptr!='\0'){
        if(*ptr == '0'){
          tone(12,440,100);
        }else{
          tone(12,440,300);
        }
        ptr++;
        delay(500);
      }
      delay(300);
     }
    }
 }
